\documentclass[a4paper]{article}

\usepackage[verbose]{xsim}

\DeclareExerciseEnvironmentTemplate{exercise}{
    \section*{\GetExerciseName\ \GetExerciseProperty{full-counter}\ (ID\ \GetExerciseProperty{id})}
}{}
\DeclareExerciseEnvironmentTemplate{subexercise}{
    \subsection*{\GetExerciseName\ \GetExerciseProperty{full-counter}\ (ID\ \GetExerciseProperty{id})}
}{}

\DeclareExerciseType{subexercise}{
    exercise-env = subexercise,
    solution-env = subsolution,
    exercise-name = subexercise,
    solution-name = subsolution,
    exercise-template = subexercise,
    solution-template = subexercise
}

\xsimsetup{
    solution/print = true,
    exercise/template = exercise,
    solution/template = exercise,
    subsolution/print = true,
    subexercise/within = exercise,
    subexercise/the-counter = \alph{subexercise}
}

\include{xsim.nested.code}

\XSIMNestedSetup{exercise}
\XSIMNestedSetup{subexercise}

\begin{document}

\begin{exercise}
    Erste Aufgabe

    \begin{subexercise}
        Erste Unteraufgabe
    \end{subexercise}
    \begin{subsolution}
        Erste Unterlösung
    \end{subsolution}

    \begin{subexercise}
        Zweite Unteraufgabe
    \end{subexercise}
    \begin{subsolution}
        Zweite Unterlösung
    \end{subsolution}

\end{exercise}
\begin{solution}
    Erste Lösung
\end{solution}

\begin{exercise}
    Zweite Aufgabe
\end{exercise}
\begin{solution}
    Zweite Lösung
\end{solution}

\end{document}
